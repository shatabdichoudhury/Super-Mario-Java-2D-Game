67         private Image getScaledImage(Image image, float x, float y) 
68         {
69     
70             // set up the transform
71             AffineTransform transform = new AffineTransform();
72             transform.scale(x, y);
73             transform.translate(
74                 (x-1) * image.getWidth(null) / 2,
75                 (y-1) * image.getHeight(null) / 2);
76     
77             // create a transparent (not translucent) image
78             Image newImage = gc.createCompatibleImage(
79                 image.getWidth(null),
80                 image.getHeight(null),
81                 Transparency.BITMASK);
82     
83             // draw the transformed image
84             Graphics2D g = (Graphics2D)newImage.getGraphics();
85             g.drawImage(image, transform, null);
86             g.dispose();
87     
88             return newImage;
89         }

131         private TileMap loadMap(String filename)
132             throws IOException
133         {
134             ArrayList lines = new ArrayList();
135             int width = 0;
136             int height = 0;
137     
138             // read every line in the text file into the list
139             BufferedReader reader = new BufferedReader(
140                 new FileReader(filename));
141             while (true) {
142                 String line = reader.readLine();
143                 // no more lines to read
144                 if (line == null) {
145                     reader.close();
146                     break;
147                 }
148     
149                 // add every line except for comments
150                 if (!line.startsWith("#")) {
151                     lines.add(line);
152                     width = Math.max(width, line.length());
153                 }
154             }
155     
156             // parse the lines to create a TileEngine
157             height = lines.size();
158             TileMap newMap = new TileMap(width, height);
159             for (int y=0; y<height; y++) {
160                 String line = (String)lines.get(y);
161                 for (int x=0; x<line.length(); x++) {
162                     char ch = line.charAt(x);
163     
164                     // check if the char represents tile A, B, C etc.
165                     int tile = ch - 'A';
166                     if (tile >= 0 && tile < tiles.size()) {
167                         newMap.setTile(x, y, (Image)tiles.get(tile));
168                     }
169     
170                     // check if the char represents a sprite
171                     else if (ch == 'o') {
172                         addSprite(newMap, coinSprite, x, y);
173                     }
174                     else if (ch == '!') {
175                         addSprite(newMap, musicSprite, x, y);
176                     }
177                     else if (ch == '*') {
178                         addSprite(newMap, goalSprite, x, y);
179                     }
180                     else if (ch == '1') {
181                         addSprite(newMap, grubSprite, x, y);
182                     }
183                     else if (ch == '2') {
184                         addSprite(newMap, flySprite, x, y);
185                     }
186                 }
187             }
188     
189             // add the player to the map
190             Sprite player = (Sprite)playerSprite.clone();
191             player.setX(TileMapDrawer.tilesToPixels(3));
192             player.setY(lines.size());
193             newMap.setPlayer(player);
194     
195             return newMap;
196         }

199         private void addSprite(TileMap map,
200             Sprite hostSprite, int tileX, int tileY)
201         {
202             if (hostSprite != null) {
203                 // clone the sprite from the "host"
204                 Sprite sprite = (Sprite)hostSprite.clone();
205     
206                 // center the sprite
207                 sprite.setX(
208                     TileMapDrawer.tilesToPixels(tileX) +
209                     (TileMapDrawer.tilesToPixels(1) -
210                     sprite.getWidth()) / 2);
211     
212                 // bottom-justify the sprite
213                 sprite.setY(
214                     TileMapDrawer.tilesToPixels(tileY + 1) -
215                     sprite.getHeight());
216     
217                 // add it to the map
218                 map.addSprite(sprite);
219             }
220         }

228         public void loadTileImages()
229         {
230             // keep looking for tile A,B,C, etc. this makes it
231             // easy to drop new tiles in the images/ directory
232             tiles = new ArrayList();
233             char ch = 'A';
234             
235             while (true) 
236             {
237                 String name = ch + ".png";
238                 File file = new File("images/" + name);
239                 if (!file.exists()) 
240                     break;
241                 
242                 tiles.add(loadImage(name));
243                 ch++;
244             }
245         }

249         public void loadCreatureSprites() 
250         {
251     
252             Image[][] images = new Image[4][];
253     
254             // load left-facing images
255             images[0] = new Image[] {
256                 loadImage("player.png"),         
257                 loadImage("fly1.png"),
258                 loadImage("fly2.png"),
259                 loadImage("fly3.png"),
260                 loadImage("grub1.png"),
261                 loadImage("grub2.png"),
262             };
263     
264             images[1] = new Image[images[0].length];
265             images[2] = new Image[images[0].length];
266             images[3] = new Image[images[0].length];
267             
268             for (int i=0; i<images[0].length; i++) 
269             {
270                 // right-facing images
271                 images[1][i] = getMirrorImage(images[0][i]);
272                 // left-facing "dead" images
273                 images[2][i] = getFlippedImage(images[0][i]);
274                 // right-facing "dead" images
275                 images[3][i] = getFlippedImage(images[1][i]);
276             }
277     
278             // create creature animations
279             Animation[] playerAnim = new Animation[4];
280             Animation[] flyAnim = new Animation[4];
281             Animation[] grubAnim = new Animation[4];
282             
283             for (int i=0; i<4; i++) 
284             {
285                 playerAnim[i] = createPlayerAnim (images[i][0]);
286                 flyAnim[i] = createFlyAnim (images[i][1], images[i][1], images[i][3]);
287                 grubAnim[i] = createGrubAnim (images[i][4], images[i][5]);
288             }
289     
290             // create creature sprites
291             playerSprite = new Player (playerAnim[0], playerAnim[1],playerAnim[2], playerAnim[3]);
292             flySprite = new Fly (flyAnim[0], flyAnim[1],flyAnim[2], flyAnim[3]);
293             grubSprite = new Grub (grubAnim[0], grubAnim[1],grubAnim[2], grubAnim[3]);
294         }
