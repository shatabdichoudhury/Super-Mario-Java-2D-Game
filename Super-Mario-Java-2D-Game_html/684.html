10     */
11     public abstract class Creature extends Sprite {

13         /**
14             Amount of time to go from STATE_DYING to STATE_DEAD.
15         */
16         private static final int DIE_TIME = 1000;
17     
18         public static final int STATE_NORMAL = 0;
19         public static final int STATE_DYING = 1;
20         public static final int STATE_DEAD = 2;

22         private Animation left;
23         private Animation right;
24         private Animation deadLeft;
25         private Animation deadRight;
26         private int state;
27         private long stateTime;

29         /**
30             Creates a new Creature with the specified Animations.
31         */
32         public Creature(Animation left, Animation right,
33             Animation deadLeft, Animation deadRight)
34         {
35             super(right);
36             this.left = left;
37             this.right = right;
38             this.deadLeft = deadLeft;
39             this.deadRight = deadRight;
40             state = STATE_NORMAL;
41         }

43     
44         public Object clone() {
45             // use reflection to create the correct subclass
46             Constructor constructor = getClass().getConstructors()[0];
47             try {
48                 return constructor.newInstance(new Object[] {
49                     (Animation)left.clone(),
50                     (Animation)right.clone(),
51                     (Animation)deadLeft.clone(),
52                     (Animation)deadRight.clone()
53                 });
54             }
55             catch (Exception ex) {
56                 // should never happen
57                 ex.printStackTrace();
58                 return null;
59             }
60         }

63         /**
64             Gets the maximum speed of this Creature.
65         */
66         public float getMaxSpeed() {
67             return 0;
68         }

70     
71         /**
72             Wakes up the creature when the Creature first appears
73             on screen. Normally, the creature starts moving left.
74         */
75         public void wakeUp() {
76             if (getState() == STATE_NORMAL && getVelocityX() == 0) {
77                 setVelocityX(-getMaxSpeed());
78             }
79         }

81     
82         /**
83             Gets the state of this Creature. The state is either
84             STATE_NORMAL, STATE_DYING, or STATE_DEAD.

86         public int getState() {
87             return state;

90     
91         /**
92             Sets the state of this Creature to STATE_NORMAL,
93             STATE_DYING, or STATE_DEAD.
94         */
95         public void setState(int state) {
96             if (this.state != state) {
97                 this.state = state;
98                 stateTime = 0;
99                 if (state == STATE_DYING) {
100                     setVelocityX(0);
101                     setVelocityY(0);
102                 }
103             }
104         }
105     
106     
107         /**
108             Checks if this creature is alive.
109         */
110         public boolean isAlive() {
111             return (state == STATE_NORMAL);
112         }
113     
114     
115         /**
116             Checks if this creature is flying.
117         */
118         public boolean isFlying() {
119             return false;
120         }

122     
123         /**
124             Called before update() if the creature collided with a
125             tile horizontally.

127         public void collideHorizontal() {
128             setVelocityX(-getVelocityX());
129         }
130     
131     
132         /**
133             Called before update() if the creature collided with a
134             tile vertically.
135         */
136         public void collideVertical() {
137             setVelocityY(0);
138         }

141         /**
142             Updates the animaton for this creature.
143         */
144         public void update(long elapsedTime) {
145             // select the correct Animation
146             Animation newAnim = anim;
147             if (getVelocityX() < 0) {
148                 newAnim = left;
149             }
150             else if (getVelocityX() > 0) {
151                 newAnim = right;
152             }
153             if (state == STATE_DYING && newAnim == left) {
154                 newAnim = deadLeft;
155             }
156             else if (state == STATE_DYING && newAnim == right) {
157                 newAnim = deadRight;
158             }
159     
160             // update the Animation
161             if (anim != newAnim) {
162                 anim = newAnim;
163                 anim.start();
164             }
165             else {
166                 anim.update(elapsedTime);
167             }
168     
169             // update to "dead" state
170             stateTime += elapsedTime;
171             if (state == STATE_DYING && stateTime >= DIE_TIME) {
172                 setState(STATE_DEAD);
173             }
174         }
