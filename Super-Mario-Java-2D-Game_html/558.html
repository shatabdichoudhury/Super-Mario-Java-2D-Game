131         /**
132          * Gets the tile that a Sprites collides with. Only the
133          * Sprite's X or Y should be changed, not both. Returns null
134          * if no collision is detected.
135          */
136         public Point getTileCollision(Sprite sprite, float newX, float newY) 
137         {
138             float fromX = Math.min(sprite.getX(), newX);
139             float fromY = Math.min(sprite.getY(), newY);
140             float toX = Math.max(sprite.getX(), newX);
141             float toY = Math.max(sprite.getY(), newY);
142             
143             // get the tile locations
144             int fromTileX = TileMapDrawer.pixelsToTiles(fromX);
145             int fromTileY = TileMapDrawer.pixelsToTiles(fromY);
146             int toTileX = TileMapDrawer.pixelsToTiles(
147                     toX + sprite.getWidth() - 1);
148             int toTileY = TileMapDrawer.pixelsToTiles(
149                     toY + sprite.getHeight() - 1);
150             
151             // check each tile for a collision
152             for (int x=fromTileX; x<=toTileX; x++) {
153                 for (int y=fromTileY; y<=toTileY; y++) {
154                     if (x < 0 || x >= map.getWidth() ||
155                             map.getTile(x, y) != null) {
156                         // collision found, return the tile
157                         pointCache.setLocation(x, y);
158                         return pointCache;
159                     }
160                 }
161             }
162             
163             // no collision found
164             return null;
165         }

168         /**
169          * Checks if two Sprites collide with one another. Returns
170          * false if the two Sprites are the same. Returns false if
171          * one of the Sprites is a Creature that is not alive.
172          */
173         public boolean isCollision(Sprite s1, Sprite s2) {
174             // if the Sprites are the same, return false
175             if (s1 == s2) {
176                 return false;
177             }
178             
179             // if one of the Sprites is a dead Creature, return false
180             if (s1 instanceof Creature && !((Creature)s1).isAlive()) {
181                 return false;
182             }
183             if (s2 instanceof Creature && !((Creature)s2).isAlive()) {
184                 return false;
185             }
186             
187             // get the pixel location of the Sprites
188             int s1x = Math.round(s1.getX());
189             int s1y = Math.round(s1.getY());
190             int s2x = Math.round(s2.getX());
191             int s2y = Math.round(s2.getY());
192             
193             // check if the two sprites' boundaries intersect
194             return (s1x < s2x + s2.getWidth() &&
195                     s2x < s1x + s1.getWidth() &&
196                     s1y < s2y + s2.getHeight() &&
197                     s2y < s1y + s1.getHeight());
198         }

201         /**
202          * Gets the Sprite that collides with the specified Sprite,
203          * or null if no Sprite collides with the specified Sprite.
204          */
205         public Sprite getSpriteCollision(Sprite sprite) {
206             
207             // run through the list of Sprites
208             Iterator i = map.getSprites();
209             while (i.hasNext()) {
210                 Sprite otherSprite = (Sprite)i.next();
211                 if (isCollision(sprite, otherSprite)) {
212                     // collision found, return the Sprite
213                     return otherSprite;
214                 }
215             }
216             
217             // no collision found
218             return null;
219         }

222         /**
223          * Updates Animation, position, and velocity of all Sprites
224          * in the current map.
225          */
226         public void update(long elapsedTime) {
227             Creature player = (Creature)map.getPlayer();
228             
229             
230             // player is dead! start map over
231             if (player.getState() == Creature.STATE_DEAD) {
232                 map = mapLoader.reloadMap();
233                 return;
234             }
235             
236             // get keyboard/mouse input
237             checkInput(elapsedTime);
238             
239             // update player
240             updateCreature(player, elapsedTime);
241             player.update(elapsedTime);
242             
243             // update other sprites
244             Iterator i = map.getSprites();
245             while (i.hasNext()) {
246                 Sprite sprite = (Sprite)i.next();
247                 if (sprite instanceof Creature) {
248                     Creature creature = (Creature)sprite;
249                     if (creature.getState() == Creature.STATE_DEAD) {
250                         i.remove();
251                     } else {
252                         updateCreature(creature, elapsedTime);
253                     }
254                 }
255                 // normal update
256                 sprite.update(elapsedTime);
257             }
258         }

261         /**
262          * Updates the creature, applying gravity for creatures that
263          * aren't flying, and checks collisions.
264          */
265         private void updateCreature(Creature creature,
266                 long elapsedTime) {
267             
268             // apply gravity
269             if (!creature.isFlying()) {
270                 creature.setVelocityY(creature.getVelocityY() +
271                         GRAVITY * elapsedTime);
272             }
273             
274             // change x
275             float dx = creature.getVelocityX();
276             float oldX = creature.getX();
277             float newX = oldX + dx * elapsedTime;
278             Point tile =
279                     getTileCollision(creature, newX, creature.getY());
280             if (tile == null) {
281                 creature.setX(newX);
282             } else {
283                 // line up with the tile boundary
284                 if (dx > 0) {
285                     creature.setX(
286                             TileMapDrawer.tilesToPixels(tile.x) -
287                             creature.getWidth());
288                 } else if (dx < 0) {
289                     creature.setX(
290                             TileMapDrawer.tilesToPixels(tile.x + 1));
291                 }
292                 creature.collideHorizontal();
293             }
294             if (creature instanceof Player) {
295                 checkPlayerCollision((Player)creature, false);
296             }
297             
298             // change y
299             float dy = creature.getVelocityY();
300             float oldY = creature.getY();
301             float newY = oldY + dy * elapsedTime;
302             tile = getTileCollision(creature, creature.getX(), newY);
303             if (tile == null) {
304                 creature.setY(newY);
305             } else {
306                 // line up with the tile boundary
307                 if (dy > 0) {
308                     creature.setY(
309                             TileMapDrawer.tilesToPixels(tile.y) -
310                             creature.getHeight());
311                 } else if (dy < 0) {
312                     creature.setY(
313                             TileMapDrawer.tilesToPixels(tile.y + 1));
314                 }
315                 creature.collideVertical();
316             }
317             if (creature instanceof Player) {
318                 boolean canKill = (oldY < creature.getY());
319                 checkPlayerCollision((Player)creature, canKill);
320             }
321             
322         }

325         /**
326          * Checks for Player collision with other Sprites. If
327          * canKill is true, collisions with Creatures will kill
328          * them.
329          */
330         public void checkPlayerCollision(Player player,
331                 boolean canKill) {
332             if (!player.isAlive()) {
333                 return;
334             }
335             
336             // check for player collision with other sprites
337             Sprite collisionSprite = getSpriteCollision(player);
338             if (collisionSprite instanceof PowerUp) {
339                 acquirePowerUp((PowerUp)collisionSprite);
340             } else if (collisionSprite instanceof Creature) {
341                 Creature badguy = (Creature)collisionSprite;
342                 if (canKill) {
343                     // kill the badguy and make player bounce
344                     badguy.setState(Creature.STATE_DYING);
345                     player.setY(badguy.getY() - player.getHeight());
346                     player.jump(true);
347                 } else {
348                     // player dies!
349                     player.setState(Creature.STATE_DYING);
350                     numLives--;
351                     if(numLives==0) {
352                         try {
353                             Thread.sleep(3000);
354                         } catch (InterruptedException ex) {
355                             ex.printStackTrace();
356                         }
357                         stop();
358                     }
359                 }
360             }
361         }
