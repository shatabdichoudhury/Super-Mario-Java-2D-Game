57             Exits the VM from a daemon thread. The daemon thread waits
58             2 seconds then calls System.exit(0). Since the VM should
59             exit when only daemon threads are running, this makes sure
60             System.exit(0) is only called if neccesary. It's neccesary

63         public void lazilyExit() {
64             Thread thread = new Thread() {
65                 public void run() {
66                     // first, wait for the VM exit on its own.
67                     try {
68                         Thread.sleep(LAZILY_CONSTANT);
69                     }
70                     catch (InterruptedException ex) { }
71                     // system is still running, so force an exit
72                     System.exit(0);
73                 }
74             };
75             thread.setDaemon(true);
76             thread.start();
77         }
