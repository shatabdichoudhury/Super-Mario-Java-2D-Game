9     /**
10         The TileMapRenderer class draws a TileMap on the screen.
11         It draws all tiles, sprites, and an optional background image
12         centered around the position of the player.
13     
14         <p>If the width of background image is smaller the width of
15         the tile map, the background image will appear to move
16         slowly, creating a parallax background effect.
17     
18         <p>Also, three static methods are provided to convert pixels
19         to tile positions, and vice-versa.
20     
21         <p>This TileMapRender uses a tile size of 64.
22     */
23     public class TileMapDrawer 
24     {
25     
26         private static final int TILE_SIZE = 64;
27         // the size in bits of the tile
28         // Math.pow(2, TILE_SIZE_BITS) == TILE_SIZE
29         private static final int TILE_SIZE_BITS = 6;
30     
31         private Image background;
32     
33         /**
34             Converts a pixel position to a tile position.
35         */
36         public static int pixelsToTiles(float pixels) {
37             return pixelsToTiles(Math.round(pixels));
38         }
39     
40     
41         /**
42             Converts a pixel position to a tile position.
43         */
44         public static int pixelsToTiles(int pixels) {
45             // use shifting to get correct values for negative pixels
46             return pixels >> TILE_SIZE_BITS;
47     
48             // or, for tile sizes that aren't a power of two,
49             // use the floor function:
50             //return (int)Math.floor((float)pixels / TILE_SIZE);
51         }
52     
53     
54         /**
55             Converts a tile position to a pixel position.
56         */
57         public static int tilesToPixels(int numTiles) {
58             // no real reason to use shifting here.
59             // it's slighty faster, but doesn't add up to much
60             // on modern processors.
61             return numTiles << TILE_SIZE_BITS;
62     
63             // use this if the tile size isn't a power of 2:
64             //return numTiles * TILE_SIZE;
65         }
66     
67     
68         /**
69             Sets the background to draw.
70         */
71         public void setBackground(Image background) {
72             this.background = background;
73         }
74     
75     
76         /**
77             Draws the specified TileMap.
78         */
79         public void draw(Graphics2D g, TileMap map,
80             int screenWidth, int screenHeight)
81         {
82             Sprite player = map.getPlayer();
83             int mapWidth = tilesToPixels(map.getWidth());
84     
85             // get the scrolling position of the map
86             // based on player's position
87             int offsetX = screenWidth / 2 -
88                 Math.round(player.getX()) - TILE_SIZE;
89             offsetX = Math.min(offsetX, 0);
90             offsetX = Math.max(offsetX, screenWidth - mapWidth);
91     
92             // get the y offset to draw all sprites and tiles
93             int offsetY = screenHeight -
94                 tilesToPixels(map.getHeight());
95     
96             // draw black background, if needed
97             if (background == null ||
98                 screenHeight > background.getHeight(null))
99             {
100                 g.setColor(Color.black);
101                 g.fillRect(0, 0, screenWidth, screenHeight);
102             }
103     
104             // draw parallax background image
105             if (background != null) {
106                 int x = offsetX *
107                     (screenWidth - background.getWidth(null)) /
108                     (screenWidth - mapWidth);
109                 int y = screenHeight - background.getHeight(null);
110     
111                 g.drawImage(background, x, y, null);
112             
113             }
114     
115             // draw the visible tiles
116             int firstTileX = pixelsToTiles(-offsetX);
117             int lastTileX = firstTileX +
118                 pixelsToTiles(screenWidth) + 1;
119             for (int y=0; y<map.getHeight(); y++) {
120                 for (int x=firstTileX; x <= lastTileX; x++) {
121                     Image image = map.getTile(x, y);
122                     if (image != null) {
123                         g.drawImage(image,
124                             tilesToPixels(x) + offsetX,
125                             tilesToPixels(y) + offsetY,
126                             null);
127                     }
128                 }
129             }
130     
131             // draw player
132             g.drawImage(player.getImage(),
133                 Math.round(player.getX()) + offsetX,
134                 Math.round(player.getY()) + offsetY,
135                 null);
136     
137             // draw sprites
138             Iterator i = map.getSprites();
139             while (i.hasNext()) {
140                 Sprite sprite = (Sprite)i.next();
141                 int x = Math.round(sprite.getX()) + offsetX;
142                 int y = Math.round(sprite.getY()) + offsetY;
143                 g.drawImage(sprite.getImage(), x, y, null);
144     
145                 // wake up the creature when it's on screen
146                 if (sprite instanceof Creature &&
147                     x >= 0 && x < screenWidth)
148                 {
149                     ((Creature)sprite).wakeUp();
150                 }
151             }
152         }
